## Operadores Lógicos

- && : and;
- || : or;
- ! : not;

Operador &&:

true && true = true;

true && false = false;

false && true = false;

false && false = false;

Operador ||;

true && true = true;

true || false = true;

false || true = true;

false || false = false;

Operador Not;

!true = false;

!false = true;
